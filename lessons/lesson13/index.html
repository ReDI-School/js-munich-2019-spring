  • Objects consist of keys and values
  • The key is like the name of the variable
  • The value is like a variable itself
  • You can access values with the dot
  • You can change values:
  • 
              let person = {
                name: "Karl",
                age: 45,
              };
              person.age = 46; // Happy birthday!

              alert(person.name + " is now " + person.age);
  • And even add new keys:
  • 
              let person = {
                name: "Karl",
                age: 45,
              };
              person.hobby = "skateboarding";

              alert(person.name + "'s hobby is " + person.hobby);
            

          let a = {
            first: 'hey',
            second: 'ho',
            space: ' ',
          };
          alert(a.first); // 1
          alert(a.first + a.space + a.second); // 2
          a.third = '!!';
          alert(a.first + a.space + a.second + a.third); // 3
        

          function lol(o, v) {
            o.b = v;
          };

          let a = {
            b: 1,
          };

          alert(a.b); // 1
          lol(a, 2);
          alert(a.b); // 2
        

          let a = {
            n: 1,
            lol: function() {
              alert(this.n);
            }
          };
          a.lol(); // 1
          alert(a.n); // 2
          a.n = 5;
          a.lol(); // 3

  • You can access single elements with []
  • Counting starts at zero!
  • You can get the total length of your array with books.length

You can add new values with .push():


            let books = [
              "Zombie Raccoons & Killer Bunnies",
              "Twilight",
              "Onio"
            ];

            books.push("The Midnight Cat Mystery Butler");

            alert(books[3]);
            alert(books.length);
          

          let a = [];
          alert(a.length); // 1
          a.push(10);
          alert(a.length); // 2
          alert(a[0]); // 3
          a.push("hello");
          alert(a.length); // 4
          alert(a[1]); // 5
